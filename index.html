clear all, clc; a=imread('pout.tif'); b = 1 double(a); [row col]=size(b); m=input('Enter the mask :'); sum=0; for i = 1 / m for j = 1 / m e = 1 input("Enter the Coefficient: '); w(i,j)=e; end end disp('mask entered by you is :'); disp(w); for d=1:m for f = 1 / m sum=sum+w(d,f); end end sum=round(sum); if(sum==0) disp('You have entered a high pass filter'); elseif (sum==1) disp('You have entered a low pass filter'); else warning('This is neither high pass nor low pass filter'); end figure.imshow(uint8(a)),title('original image'); s = (m + 1) / 2 for x=1:row for :col y = 1 b(x,a(x,y); end end for x=s:row-s for y=s:col-s end b(x, y) = 0 end for x=s:row-s for y=s:col-s for i=1:m for j=1:m c=double(w(i,j)); d=double(a(x-s+i,y-s+j)); f=c.*d; b(x,y)=b(x,y)+f; end end end end if(sum==0) figure,imshow(uint8(b)),title('High pass filter'); elseif (sum==1) figure,imshow(uint8(b)),title('Low pass filter'); else figure,imshow(uint8(b)), title('Filtered Image'); end
